  console.log('| Repository | Skill | Description |');

  for (const [repoKey, repo] of Object.entries(catalog.repos || {})) {
    if (!repo.skills) continue;
    for (const [skillKey, skill] of Object.entries(repo.skills)) {
      const desc = (skill.description || 'No description available')
        .replace(/\|/g, '∣')
        .replace(/[\r\n]+/g, ' ')
        .replace(/\\/g, '╲')
        .trim();
      console.log(`| ${repoKey} | =${skillKey}= | ${desc} |`);
    }
  }
